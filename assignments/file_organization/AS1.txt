#include <iostream>

// Function to read file till EOF using eof()
void ex1readFileWordByWord(string fileName) {
    fstream file;
    file.open(fileName, ios::in);
    if (file.good()) {
        string l;
        cout << "----- File Content -----\n";
        while (!file.eof()) {
            getline(file, l);
            cout << l << endl;
        }
    }
    file.close();

}

// Function to count all characters using get(ch)
void ex1countCharacters(string fileName) {
    ifstream file;
    file.open(fileName);

    if (file) {
        char c;

        int i = 0;
        while (!file.eof()) {
            file >> c;
            cout << c << "(" << i << ")" << " ";
            i++;
        }
        cout << endl << "total words = " << i << endl;

        file.close();
    }
    file.close();

}

void ex2writeToFile() {
    fstream file;
    file.open("file1.txt", ios::out); // Open file for writing


    if (file.good()) {
        char ch;
        cout << "Enter your text (type '.' to end): " << endl;

        // Read character by character until user enters '.'
        while (true) {
            cin.get(ch);
            file.put(ch);
            if (ch == '.')
                break;
        }


    }
    file.close();
}
